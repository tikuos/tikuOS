TikuOS ships with 11 complete examples that progress from a basic LED blink to sensor pipelines and channel-based message passing. Each example is a single C file that compiles as a standalone firmware image.

Building Examples

When no APP= flag is specified, examples are auto-detected from the examples/ directory:

make MCU=msp430fr5969
Note: Only one of apps, tests, or examples can be active at a time. When APP= is specified, examples are automatically excluded from compilation.

Example 01: Blink

What You Learn

  • Declaring and defining a process with TIKU_PROCESS
  • Using TIKU_AUTOSTART_PROCESSES for automatic startup
  • Setting an event timer and waiting for TIKU_EVENT_TIMER
  • Drift-free periodic scheduling with tiku_timer_reset()

Code

#include "tiku.h"

/* Declare and define the process */
TIKU_PROCESS(blink_process, "Blink");

/* Timer structure (must be static -- persists across yields) */
static struct tiku_timer blink_timer;

TIKU_PROCESS_THREAD(blink_process, ev, data)
{
    TIKU_PROCESS_BEGIN();

    /* One-time initialization */
    tiku_common_led1_init();

    /* Start a 1-second periodic timer */
    tiku_timer_set_event(&blink_timer, TIKU_CLOCK_SECOND);

    while (1) {
        /* Block until the timer fires */
        TIKU_PROCESS_WAIT_EVENT_UNTIL(ev == TIKU_EVENT_TIMER);

        /* Toggle the LED */
        tiku_common_led1_toggle();

        /* Reset the timer (drift-free: next = previous + interval) */
        tiku_timer_reset(&blink_timer);
    }

    TIKU_PROCESS_END();
}

/* Register for automatic startup */
TIKU_AUTOSTART_PROCESSES(&blink_process);

Key Points

TIKU_PROCESS(blink_process, "Blink") simultaneously declares the process thread function and defines a tiku_process_t struct. The timer struct must be static because it persists across protothread yields. tiku_timer_reset() anchors the next expiration to the previous start time (start += interval), avoiding cumulative drift. The TIKU_PROCESS_WAIT_EVENT_UNTIL macro yields the protothread until the specified condition is true.

Example 04: Multi-Process

What You Learn

  • Posting custom events to a specific process
  • Receiving events and extracting data
  • Defining custom event IDs starting from TIKU_EVENT_USER
  • Coordinating two processes

Code

#include "tiku.h"

/* Custom event ID */
#define EVENT_NEW_DATA    (TIKU_EVENT_USER + 0)

/* Declare two processes */
TIKU_PROCESS(producer, "Producer");
TIKU_PROCESS(consumer, "Consumer");

/* Producer -- sends counter value every 2 seconds */
static struct tiku_timer producer_timer;

TIKU_PROCESS_THREAD(producer, ev, data)
{
    static int counter = 0;

    TIKU_PROCESS_BEGIN();

    tiku_common_led2_init();
    tiku_timer_set_event(&producer_timer, TIKU_CLOCK_SECOND * 2);

    while (1) {
        TIKU_PROCESS_WAIT_EVENT_UNTIL(ev == TIKU_EVENT_TIMER);

        /* Post data to the consumer */
        tiku_process_post(&consumer, EVENT_NEW_DATA,
                          (void *)(intptr_t)counter);
        counter++;

        tiku_common_led2_toggle();
        tiku_timer_reset(&producer_timer);
    }

    TIKU_PROCESS_END();
}

/* Consumer -- receives data and acts on it */
TIKU_PROCESS_THREAD(consumer, ev, data)
{
    TIKU_PROCESS_BEGIN();

    tiku_common_led1_init();

    while (1) {
        TIKU_PROCESS_WAIT_EVENT_UNTIL(ev == EVENT_NEW_DATA);

        /* Extract the counter value */
        int value = (int)(intptr_t)data;
        (void)value;  /* Use it -- e.g., threshold check */

        /* Indicate reception */
        tiku_common_led1_toggle();
    }

    TIKU_PROCESS_END();
}

/* Autostart both */
TIKU_AUTOSTART_PROCESSES(&producer, &consumer);

Key Points

Custom event IDs start at TIKU_EVENT_USER (0x10) to avoid collisions with system events. tiku_process_post(&consumer, EVENT_NEW_DATA, ...) enqueues an event specifically for the consumer process. The data pointer carries the counter value cast to void *. The consumer extracts it with a reverse cast. Both processes are autostarted together.

Example 06: Callback Timer

What You Learn

  • Setting callback-mode timers with tiku_timer_set_callback()
  • Rescheduling inside the callback for periodic behavior
  • Comparing callback timers vs event timers

Code

#include "tiku.h"

static struct tiku_timer timer_a;
static struct tiku_timer timer_b;

/* Callbacks */
static void on_timer_a(void *ptr)
{
    (void)ptr;
    tiku_common_led1_toggle();
    tiku_timer_reset(&timer_a);  /* Periodic */
}

static void on_timer_b(void *ptr)
{
    (void)ptr;
    tiku_common_led2_toggle();
    tiku_timer_reset(&timer_b);  /* Periodic */
}

/* Process -- sets up timers then idles */
TIKU_PROCESS(callback_demo, "Callback Demo");

TIKU_PROCESS_THREAD(callback_demo, ev, data)
{
    TIKU_PROCESS_BEGIN();

    tiku_common_led1_init();
    tiku_common_led2_init();

    /* Start callback timers */
    tiku_timer_set_callback(&timer_a, TIKU_CLOCK_SECOND / 2,
                            on_timer_a, NULL);
    tiku_timer_set_callback(&timer_b,
                            TIKU_CLOCK_SECOND + TIKU_CLOCK_SECOND * 3 / 10,
                            on_timer_b, NULL);

    /* Nothing else to do -- callbacks handle everything */
    while (1) {
        TIKU_PROCESS_WAIT_EVENT();
    }

    TIKU_PROCESS_END();
}

TIKU_AUTOSTART_PROCESSES(&callback_demo);

Key Points

Callback timers call a function directly when they expire, without posting an event to a process. This is useful for simple periodic actions that do not need a full process context. Timer A toggles LED1 every 500 ms; Timer B toggles LED2 every 1.3 seconds. Both reschedule themselves by calling tiku_timer_reset() inside the callback. The main process thread just idles after setup, as callbacks handle all the work.

Example 09: Channel

What You Learn

  • Declaring a type-safe channel with TIKU_CHANNEL_DECLARE
  • Putting and getting typed messages
  • Combining channels with events to wake a consumer
  • Querying channel state (empty / free slots)

Code

#include "tiku.h"

/* Message type */
struct sensor_msg {
    uint16_t seq;       /* Sequence number */
    uint16_t value;     /* Simulated sensor value */
};

/* Type-safe channel: 4 messages deep */
TIKU_CHANNEL_DECLARE(sensor_ch, struct sensor_msg, 4);

#define EVENT_DATA_READY   (TIKU_EVENT_USER + 0)

TIKU_PROCESS(sensor_proc, "Sensor");
TIKU_PROCESS(display_proc, "Display");

/* Sensor -- produces a reading every second */
static struct tiku_timer sensor_timer;

TIKU_PROCESS_THREAD(sensor_proc, ev, data)
{
    static uint16_t seq = 0;

    TIKU_PROCESS_BEGIN();

    tiku_common_led2_init();
    sensor_ch_init();
    tiku_timer_set_event(&sensor_timer, TIKU_CLOCK_SECOND);

    while (1) {
        TIKU_PROCESS_WAIT_EVENT_UNTIL(ev == TIKU_EVENT_TIMER);

        struct sensor_msg msg;
        msg.seq   = seq++;
        msg.value = seq * 10;

        if (sensor_ch_put(&msg)) {
            tiku_process_post(&display_proc, EVENT_DATA_READY, NULL);
            tiku_common_led2_toggle();
        }

        tiku_timer_reset(&sensor_timer);
    }

    TIKU_PROCESS_END();
}

/* Display -- drains all buffered messages when notified */
TIKU_PROCESS_THREAD(display_proc, ev, data)
{
    TIKU_PROCESS_BEGIN();

    tiku_common_led1_init();

    while (1) {
        TIKU_PROCESS_WAIT_EVENT_UNTIL(ev == EVENT_DATA_READY);

        struct sensor_msg msg;
        while (sensor_ch_get(&msg)) {
            tiku_common_led1_toggle();
        }
    }

    TIKU_PROCESS_END();
}

TIKU_AUTOSTART_PROCESSES(&sensor_proc, &display_proc);

Key Points

TIKU_CHANNEL_DECLARE(sensor_ch, struct sensor_msg, 4) generates static storage, a channel instance, and typed helpers: sensor_ch_init(), sensor_ch_put(), and sensor_ch_get(). The compiler will reject pointer type mismatches at call sites. Channels decouple producers and consumers: the producer never needs to know which process will read the data. The consumer drains all buffered messages in one go when it receives the notification event.
